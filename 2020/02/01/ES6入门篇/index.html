        var _this = this;

    throw new Errror('can\'t be empty')

可变参数

//es3 es5 可变参数
function f(){
    var a=Array.prototype.slice.call(arguments);
    var sum=0;
    a.forEach(function(item){
        sum+=item*1;
    })
    return sum
}
console.log(f(1,2,3,6));

//es6 可变参数
function f(...a){
    var sum=0;
    a.forEach(item => {
        sum+=item*1
    });
    return sum
}
console.log(f(1,2,3,6));

//es5合并数组
var params=['hello',true,7];
var other=[1,2].concat[params];
console.log(other);

//es6合并数组
var params['hello',true,7];
var other=[1,2,...params];
console.log(other);
